class JkadLatency2:
    patt = re.compile(".*#\*LATENCY\s*(\d*).*")
    def __init__(self, l_limit, u_limit):
        self.l_limit = float(l_limit)
        self.u_limit = float(u_limit)

    def isOK(self, mkid):
        return True if mkid > self.l_limit and mkid < self.u_limit else False

    def get_la_map(self, path):
        lamap = {}
        mkid = 1
        with open(path) as f:
            for line in f:
                result = JkadLatency2.patt.match(line)
                if result:
                    lamap[mkid] = int(result.group(1))
                    mkid += 1
        return {k: lamap[k] for k in lamap if self.isOK(k)}

class DataAnalysis:
    def __init__(self, logfile, l_limit = 150, u_limit = 'Inf', la_type = "ave", mode = 2):
        self.logfile = logfile
        self.l_limit = l_limit
        self.u_limit = u_limit
        self.la_type = la_type
        # mode:
        # 1 (log: use 'FROMSOURCE' and 'FROMFINAL' marker)
        # 2 (log: use 'LATENCY' marker)
        self.mode = mode

    @staticmethod
    def get_la(lmap, la_type="ave"):
        if la_type in ["99", "97", "95", "90", "80", "75", "50", "25", "0"]:
            return np.percentile(np.array(lmap.values()), int(la_type))
        else:
            return np.average(np.array(lmap.values()))
    
    @staticmethod
    def show_map(lmap, figure_name="figure", la_type="ave"):
        if lmap == {}:
            print("Empty data set!")
            return
        title_str = ("{}, {} Latency: {:.2f}ms").format(
                    figure_name, la_type, DataAnalysis.get_la(lmap, la_type))
        fig = plt.figure(figsize=(18, 6))
        fig.suptitle(title_str, size = 20)
        plt.plot(sorted(lmap), [lmap[k] for k in sorted(lmap)])
        plt.xlabel("MarkerId", fontsize = 20, color = 'purple')
        plt.ylabel("Latency (ms)", fontsize = 20, color = 'purple')
        plt.show()
        
        df = pd.DataFrame(lmap.items(), columns=["markerid", "latency"])
        g = sns.FacetGrid(df, size = 4, aspect = 2)
        g.map(sns.distplot, "latency")

    # get current dataset's latency
    def get_myla(self):
        return DataAnalysis.get_la(self.get_la_map(), self.la_type)
    
    def get_la_map(self):
        if self.mode == 1:
            return JkadLatency(self.l_limit, self.u_limit).get_la_map(self.logfile)
        else:
            return JkadLatency2(self.l_limit, self.u_limit).get_la_map(self.logfile)
    
    def show(self):
        lmap = self.get_la_map()
        DataAnalysis.show_map(lmap, self.logfile, self.la_type)

SLEEPPATT = re.compile(r".*negative sleeptime: ([-]\d*)ms.*")
SOURCEPATT = re.compile(r".*#\*FROMSOURCE\s*([-]\d*)\s*(\d*).*")
startmap = {}; deltamap = {}; sleepmap = {}
markercnt = 1; sleepcnt = 1
with open("LOGS/20161222/5000-3000/3004.log") as f:
    for line in f:
        res = SLEEPPATT.match(line)
        if res:
            sleepmap[sleepcnt] = int(res.group(1))
            sleepcnt += 1
        res = SOURCEPATT.match(line)
        if res:
            startmap[markercnt] = int(res.group(2))
            markercnt += 1
            
for k in range(1, markercnt-1):
    deltamap[k] = startmap[k+1] - startmap[k]
    
DataAnalysis.show_map(deltamap, figure_name="deltamap")
DataAnalysis.show_map(sleepmap, figure_name="sleeptime")
